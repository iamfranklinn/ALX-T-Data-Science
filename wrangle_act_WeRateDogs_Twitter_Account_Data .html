def fix_source(x):
    'x is an html string from the source column in df_arch_cleaned dataset'
    # Finding the first closed  tag >
    i= x.find('>') + 1
    # Finding the first open tag after the previous <
    j =x[i:].find('<')
    # Extracting the text in between
    return x[i:][:j]
